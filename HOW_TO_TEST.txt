STEP 2: TEST WITH DOCKER (RECOMMENDED - 10 minutes)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

This creates a real Prometheus + Node Exporter environment with sample data.

# 1. Create test directory
mkdir -p /tmp/prom-test
cd /tmp/prom-test

# 2. Create docker-compose.yml
cat > docker-compose.yml << 'DOCKEREOF'
version: '3'
services:
  prometheus:
    image: prom/prometheus:latest
    volumes:
      - ./prometheus.yml:/etc/prometheus/prometheus.yml
      - prometheus-data:/prometheus
    ports:
      - "9090:9090"
  node-exporter:
    image: prom/node-exporter:latest
    ports:
      - "9100:9100"
volumes:
  prometheus-data:
DOCKEREOF

# 3. Create Prometheus config
cat > prometheus.yml << 'PROMEOF'
global:
  scrape_interval: 15s
scrape_configs:
  - job_name: 'node'
    static_configs:
      - targets: ['node-exporter:9100']
PROMEOF

# 4. Start services
docker-compose up -d

# 5. Wait for data collection (5 minutes)
echo "Collecting metrics... waiting 5 minutes"
for i in {1..300}; do
  echo -ne "Progress: $i/300 seconds\r"
  sleep 1
done
echo ""

# 6. Extract data from Docker volume
VOLUME=$(docker volume ls -q | grep prom-test_prometheus-data)
echo "Found volume: $VOLUME"

docker run --rm \
  -v ${VOLUME}:/source \
  -v $(pwd)/data:/target \
  alpine sh -c "cp -r /source/* /target/"

echo "Data extracted to: $(pwd)/data"
ls -lh data/

# 7. Stop containers (cleanup)
docker-compose down

# 8. Test prom2sar
cd /home/ssonigra/reprod/claude/prometheus-dump-operator

./bin/prom2sar \
  -tsdb /tmp/prom-test/data \
  -output /tmp/prom-test/results \
  -verbose

# 9. View results
echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "SUMMARY OUTPUT:"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
cat /tmp/prom-test/results/sar-summary-*.txt

echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "FULL REPORT (first 50 lines):"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

# Search for specific time

grep "12:00:00" sar-*.txt

# Export all data points for a specific time window
grep -E "12:[0-5][0-9]:" sar-*.txt > noon-hour-data.txt

Scenario: Application had issues on June 12 between 10am and 12pm

# 1. Convert only the incident time window
./bin/prom2sar \
  -tsdb /var/lib/prometheus/data \
  -start 2026-06-12T10:00:00Z \
  -end 2026-06-12T12:00:00Z \
  -output ./incident-2026-06-12 \
  -profile all \
  -verbose

cd incident-2026-06-12

# 2. Check overall summary
cat sar-summary-*.txt

# 3. Find CPU spikes during incident
grep -A 2 "CPU" sar-*.txt | \
  grep -v "all" | \
  awk '{if ($2 > 80) print "CPU spike at", $1, "-", $2"%"}'

# 4. Check memory at exact incident time (e.g., 10:45am)
echo "Memory at incident time:"
grep "10:45" sar-*.txt | grep -A 5 "Memory"

# 5. Check disk I/O at incident time
echo "Disk I/O at incident time:"
grep -A 10 "Disk" sar-*.txt | grep "10:45"

# 6. Check network activity
echo "Network activity:"
grep -A 10 "Network" sar-*.txt | grep "10:45"

# 7. Package for sharing with team
tar czf incident-sar-data.tar.gz sar-*.txt
echo "Created: incident-sar-data.tar.gz"

TROUBLESHOOTING
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Problem: "Error: TSDB path does not exist"
  → Check the path is correct
  → Ensure you have read permissions
  → Verify it's a Prometheus data directory with blocks

Problem: "No data found in the specified time range"
  → Check TSDB blocks time ranges with -verbose
  → Adjust start/end times to match available data
  → Ensure Prometheus collected data during that period

Problem: Binary not found
  → Run: make build-cli
  → Check: ls -lh bin/prom2sar

Problem: Permission denied
  → Run: chmod +x bin/prom2sar
  → Or use: sudo ./bin/prom2sar ...
